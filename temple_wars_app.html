    <script>

        const allHeroes = [

            'Aang', 'Amon', 'Asami Sato', 'Azula', 'Borte', 'Bumi', 'Katara (PL)', 'Katara (WBM)', 

            'King Bumi', 'Korra', 'Kuei', 'Kuvira', 'Kyoshi', 'Lin Beifong', 'Mako', 'Meelo', 

            'Piandao', 'Roku', 'Sokka', 'Suki', 'Tenzin', 'Teo', 'Toph', 'Unalaq', 'Uncle Iroh', 

            'Yangchen', 'Yue', 'Zuko'

        ];


        const heroSelectionGrid = document.getElementById('hero-selection-grid');

        allHeroes.forEach(hero => {

            const div = document.createElement('div');

            div.className = 'flex items-center p-2 bg-gray-50 rounded-md border border-gray-200';

            const checkbox = document.createElement('input');

            checkbox.type = 'checkbox';

            checkbox.id = hero;

            checkbox.value = hero;

            checkbox.className = 'w-5 h-5 text-blue-600 bg-gray-100 border-gray-300 rounded focus:ring-blue-500';

            const label = document.createElement('label');

            label.htmlFor = hero;

            label.textContent = hero;

            label.className = 'ml-3 text-sm font-medium text-gray-800';

            div.appendChild(checkbox);

            div.appendChild(label);

            heroSelectionGrid.appendChild(div);

        });


        const generateBtn = document.getElementById('generateBtn');

        const resultsContainer = document.getElementById('results-container');

        const resultsOutput = document.getElementById('results-output');

        const loader = document.getElementById('loader');


        generateBtn.addEventListener('click', async () => {

            const selectedHeroes = Array.from(heroSelectionGrid.querySelectorAll('input[type=checkbox]:checked')).map(cb => cb.value);


            if (selectedHeroes.length < 3) {

                resultsContainer.classList.remove('hidden');

                resultsOutput.innerHTML = `<p class="text-red-500 text-center font-semibold">Please select at least 3 heroes to form a team.</p>`;

                return;

            }


            resultsContainer.classList.remove('hidden');

            resultsOutput.innerHTML = '';

            loader.classList.remove('hidden');


            const systemPrompt = "You are an expert strategist for the game 'Avatar: Realms Collide'. You will be given a list of owned heroes and must create optimized 3-person teams for Temple War battles. Your response must be in JSON format.";

            const userPrompt = `

                My owned heroes are: ${selectedHeroes.join(', ')}.


                Based ONLY on the heroes from this list, generate at least five 3-person team combinations that are optimized for Temple War battles. 

                For each team, provide a name, the composition, and a brief strategy explaining why it's effective for either offense (capturing a temple) or defense (holding a temple).


                Your response must be a single JSON object with a key "teams", which is an array of objects. Each object in the array must have these keys: "teamName", "composition", and "strategy".

            `;


            try {

                const apiKey = ""; // IMPORTANT: REPLACE WITH YOUR OPENAI API KEY

                const apiUrl = "https://api.openai.com/v1/chat/completions";

                

                const payload = {

                    model: "gpt-3.5-turbo-1106", // A model that supports JSON mode

                    response_format: { "type": "json_object" },

                    messages: [

                        { "role": "system", "content": systemPrompt },

                        { "role": "user", "content": userPrompt }

                    ]

                };


                const response = await fetch(apiUrl, {

                    method: 'POST',

                    headers: { 

                        'Content-Type': 'application/json',

                        'Authorization': `Bearer ${apiKey}`

                    },

                    body: JSON.stringify(payload)

                });


                if (!response.ok) {

                    throw new Error(`API request failed with status ${response.status}`);

                }


                const result = await response.json();

                

                if (result.choices && result.choices.length > 0 && result.choices[0].message) {

                    const content = result.choices[0].message.content;

                    const parsedJson = JSON.parse(content);

                    

                    if (parsedJson.teams && parsedJson.teams.length > 0) {

                        displayResults(parsedJson.teams);

                    } else {

                        throw new Error("No teams were generated in the response.");

                    }

                } else {

                    throw new Error("Invalid response structure from API.");

                }


            } catch (error) {

                console.error("Error fetching team recommendations:", error);

                resultsOutput.innerHTML = `<p class="text-red-500 text-center font-semibold">An error occurred while generating teams. Please check your API key and try again.</p>`;

            } finally {

                loader.classList.add('hidden');

            }

        });


        function displayResults(teams) {

            let html = '';

            teams.forEach(team => {

                html += `

                    <div class="result-card">

                        <h3>${team.teamName}</h3>

                        <h4>Composition: ${team.composition.join(', ')}</h4>

                        <div class="prose prose-sm max-w-none mt-2">

                           ${marked.parse(team.strategy)}

                        </div>

                    </div>

                `;

            });

            resultsOutput.innerHTML = html;

        }
